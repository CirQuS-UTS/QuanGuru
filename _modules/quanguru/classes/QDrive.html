[docs]    def integrateShape(self, timePoints):
        integral = 0
        for ind in range(len(timePoints)-1):
            integral += self.apply((timePoints[ind+1]+timePoints[ind])/2)*(timePoints[ind+1]-timePoints[ind])
        return integral
